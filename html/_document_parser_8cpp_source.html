17  continue;

21 
22  char readBuffer[65536];

24  rapidjson::Document d;

30  rapidjson::Value &authors = d["metadata"]["authors"];
31  std::string text;
32  for(int i = 0; i < abstract.Size(); i++){
33  if(abstract[i].HasMember("text")){
34  text += abstract[i]["text"].GetString();
35  text += " ";
36  }
37  }
38  for(int i = 0; i < body.Size(); i++){
39  if(body[i].HasMember("text")){
40  text += body[i]["text"].GetString();
41  text += " ";
42  }
43  }
44  std::vector<std::string> author_last;
45  for(int i = 0; i < authors.Size(); i++){
46  if(authors[i].HasMember("last")){
47  author_last.push_back(authors[i]["last"].GetString());

50  ResearchDocument doc(id.GetString(), title.GetString(), text, author_last);
