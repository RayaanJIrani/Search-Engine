254 inline void DSAVLTree<A, T>::balance(DSAVLTreeNode<A, T> *&t) {
255  if(getHeight(t->left) - getHeight(t->right) > 1){
256  if(getHeight(t->left->left) >= getHeight(t->left->right)){
257  rotateWithLeftChild(t);

265  doubleWithRightChild(t);

269  rotateWithRightChild(t);

271  }
